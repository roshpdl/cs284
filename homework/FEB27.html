 select
    match_time,
    p1.display_name as human,
    p2.display_name as bot,
    d1.throw_name as human_throw,
    d2.throw_name as bot_throw
from
    rps_matches as m,
    rps_players as p1,
    rps_players as p2,
    rps_throw_descriptions as d1,
    rps_throw_descriptions as d2
where
    m.player_1_id = p1.player_id
    and m.player_2_id = p2.player_id
    and m.player_1_throw = d1.throw_abb
    and m.player_2_throw = d2.throw_abb
    and DATE(match_time) = "2023-02-18";
